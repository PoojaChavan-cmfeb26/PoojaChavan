ASSIGNMENT 6
) Count all orders for October 3
SELECT COUNT(*) AS total_orders
FROM Orders
WHERE Odate = '2026-10-03';

2) Count the number of different non-NULL city values in Customers
SELECT COUNT(DISTINCT city) AS distinct_cities
FROM Customer
WHERE city IS NOT NULL;

3) Select each customer’s smallest order
SELECT customer_id, MIN(Amt) AS smallest_order
FROM Orders
GROUP BY customer_id;

4) Select the first customer, alphabetically, whose name begins with G
SELECT customer_name
FROM Customers
WHERE customer_name LIKE 'G%'
ORDER BY customer_name ASC
FETCH FIRST 1 ROW ONLY;

5) Select the highest rating in each city
SELECT city, MAX(rating) AS highest_rating
FROM Customers
GROUP BY city;

6) Count the number of salespeople registering orders for each day
(ensuring each salesperson As counted only once per day)
SELECT order_date, COUNT(DISTINCT salesperson_id) AS salespeople_count
FROM Orders
GROUP BY order_date;
